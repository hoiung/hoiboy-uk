{{- /* Breadcrumbs (site structure, not click history). Posts (with a category) walk via the FIRST category tag, not the section. Everything else walks its section ancestry via .Ancestors. A page may set `breadcrumbParent` (a page path) to sit under a page other than its Hugo section, e.g. the form thanks page under the AGIT page. Hidden on home. */ -}} {{- if not .IsHome -}}
Home {{- if and .IsPage (index (.Params.categories | default slice) 0) -}} {{- /* Posts: derive the middle crumb from the first category param. Look up the section page so the label comes from content//_index.md. */ -}} {{- $slug := index .Params.categories 0 -}} {{- $sectionPage := site.GetPage (printf "/%s" $slug) -}} {{- $href := printf "/%s/" ($slug | urlize) -}} {{- $label := $slug | title -}} {{- with $sectionPage -}} {{- $href = .RelPermalink -}} {{- $label = .Title -}} {{- end -}} >{{ $label }} >{{ .Title }} {{- else -}} {{- /* Walk the ancestry. An explicit breadcrumbParent override replaces this page's own ancestry with the named parent's ancestry plus that parent. */ -}} {{- $trail := .Ancestors.Reverse -}} {{- $extra := false -}} {{- with .Params.breadcrumbParent -}} {{- with site.GetPage . -}} {{- $trail = .Ancestors.Reverse -}} {{- $extra = . -}} {{- end -}} {{- end -}} {{- range $trail -}} {{- if not .IsHome -}} >{{ .Title }} {{- end -}} {{- end -}} {{- with $extra -}} >{{ .Title }} {{- end -}} >{{ .Title }} {{- end -}}
{{- end -}}
